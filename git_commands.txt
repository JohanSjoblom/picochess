# Comandos para subir solo la carpeta web a GitHub

# 1. Navegar al directorio del proyecto
cd "d:\Python\picochess-Study-Position"

# 2. Inicializar git si no está inicializado
git init

# 3. Agregar el remote de la rama Study-Position
git remote add origin https://github.com/JohanSjoblom/picochess.git

# 4. Hacer fetch para obtener información de ramas remotas
git fetch origin

# 5. Cambiar a la rama Study-Position remota
git checkout -b Study-Position origin/Study-Position

# 6. Agregar solo la carpeta web
git add web/

# 7. Hacer commit con mensaje descriptivo
git commit -m "Fix JavaScript errors in webserver

- Fixed setHeaders function to validate data object before using 'in' operator
- Added global error handling for Bootstrap null value errors  
- Silenced AJAX connection errors for auxiliary servers (books/games)
- Added input validation patches to prevent split() errors on null values"

# 8. Subir los cambios
git push origin Study-Position
